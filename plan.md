### Python
```python
def greet(name):
    """Greets the user."""
    greeting = f"Hello, {name}!"
    print(greeting)

if __name__ == "__main__":
    greet("Retro User")
```

### C++
```cpp
#include <iostream>
#include <vector>

int main() {
    std::vector<std::string> msg = {"Hello", "C++", "World"};
    for (const auto& word : msg) {
        std::cout << word << " ";
    }
    std::cout << std::endl;
    return 0;
}
```
